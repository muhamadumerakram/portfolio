export const generateRandomString = (length = 11) => {
  const randomNumber = Math.random()
  const randomString = randomNumber.toString(36).substring(2)
 
  if (randomString.length < length) {
    return randomString.padEnd(length, '0')
  }
  return randomString.slice(0, length)
}

Test Case Function 

import { interceptGqlRequest, generateRandomString} from "../../utils/helpers"

// ApiKey class for API key management test cases
class ApiKey {
  constructor() {
    this.verifyAPIRoute()
  }
  // Navigates to the API Keys settings page
  verifyAPIRoute(){
    cy.visit('/sites')
    cy.get("[data-attribute='dropdown-user-profile']").click()
    cy.get("[data-attribute='dropdown-item-user-account']").click()
    cy.get("[data-attribute='setting-card-api-keys']").click()
  }

  // Generates a new API key and verifies the process
  verifyGenerateApi(){
    interceptGqlRequest({ alias: 'generateApiKey' })
    cy.get("[data-attribute='button-generate-api-key']").click() // click on generate api button
    cy.get("[data-attribute='input-api-name']").type(generateRandomString()) // type api key name
    cy.get("[data-attribute= 'button-modal-save-api-key']").click() // click on save button
    cy.wait('@generateApiKey')
  }

  // Deletes an API key and verifies the process
  verifyDeleteApi(){
    interceptGqlRequest({ alias: 'deleteApiKey' })
    cy.get("[data-attribute='dropdown-menu-button']").first().click()
    cy.get("[data-attribute='dropdown-item-api-delete']").click({ waitForAnimations: true})
    cy.get("[data-attribute='button-delete-api-key']").click()
    cy.wait('@deleteApiKey')
  }

  // Deactivates an API key and verifies the process
  verifyDeactivateApi(){
    interceptGqlRequest({ alias: 'updateApiKey' })
    cy.get("[data-attribute='dropdown-menu-button']").first().click()
    cy.get("[data-attribute='dropdown-item-api-deactivate']").click()
    cy.wait('@updateApiKey')    
  }

  // Activates an API key and verifies the process
  verifyActivateApi(){
    interceptGqlRequest({ alias: 'updateApiKey' })
    cy.get("[data-attribute='dropdown-menu-button']").first().click()
    cy.get("[data-attribute='dropdown-item-api-deactivate']").click()
    cy.wait('@updateApiKey')
  }
}
export default ApiKey
